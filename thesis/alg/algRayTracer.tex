    \caption{Ray Tracing Algorithm Overview}
    \label{alg:ray}
    \begin{algorithmic}
        \ForEach{Source $S_i \in S$}
            \ForEach{Receiver $R_j \in R$}
            \State $G_{ij} \gets \emptyset$
                \ForEach{Order $l \in \{0...L\}$}
                    \State $P \gets$ AddPaths($S_i$, $R_j$, $L$)
                    \If{$P \neq \emptyset$}
                    \State $G_{ij} = G_{ij} \cup P$
                    \EndIf
                \EndFor
            \EndFor
        \EndFor
    \end{algorithmic}
\end{algorithm}
